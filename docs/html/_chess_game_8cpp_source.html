19void ChessGame::resetToStartingPosition() {
20 pieces_.clear();
21 board_.fill({});
22 currentMoveIndex_ = 0;
23 autoPlayAccum_ = 0.0f;
24
25 auto add = [this](PieceType t, Color c, int file, int rank) {
26 auto p = std::make_unique<ChessPiece>(t, c, Square{file, rank});
27 placePiece(p.get(), {file, rank});
28 pieces_.push_back(std::move(p));
29 };
30
31 // Pawns
32 for (int f = 0; f < 8; ++f) {
33 add(PieceType::Pawn, Color::White, f, 1);
34 add(PieceType::Pawn, Color::Black, f, 6);
35 }
36 // Major pieces
37 const PieceType backRank[8] = {
38 PieceType::Rook, PieceType::Knight, PieceType::Bishop, PieceType::Queen,
39 PieceType::King, PieceType::Bishop, PieceType::Knight, PieceType::Rook
40 };
41 for (int f = 0; f < 8; ++f) {
42 add(backRank[f], Color::White, f, 0);
43 add(backRank[f], Color::Black, f, 7);
44 }
45}

61bool ChessGame::canPieceReach(const ChessPiece& p, Square to, bool isCapture) const {
62 if (!to.valid()) return false;
63 Square from = p.square();
64 if (from == to) return false;
65 int df = to.file - from.file;
66 int dr = to.rank - from.rank;
67 int adf = std::abs(df);
68 int adr = std::abs(dr);
69
70 auto rayClear = [&](int sf, int sr) {
71 Square cur{ from.file + sf, from.rank + sr };
72 while (cur != to) {
73 if (pieceAt(cur)) return false;
74 cur.file += sf;
75 cur.rank += sr;
76 }
77 return true;
78 };
79
80 switch (p.type()) {
81 case PieceType::Knight:
82 return (adf == 1 && adr == 2) || (adf == 2 && adr == 1);
83 case PieceType::King:
84 return adf <= 1 && adr <= 1 && (adf + adr > 0);
85 case PieceType::Pawn: {
86 int dir = (p.color() == Color::White) ? 1 : -1;
87 if (isCapture) {
88 return adf == 1 && dr == dir;
89 }
90 if (df != 0) return false;
91 if (dr == dir) return pieceAt(to) == nullptr;
92 if (dr == 2 * dir) {
93 int startRank = (p.color() == Color::White) ? 1 : 6;
94 if (from.rank != startRank) return false;
95 Square mid{ from.file, from.rank + dir };
96 return pieceAt(mid) == nullptr && pieceAt(to) == nullptr;
97 }
98 return false;
99 }
100 case PieceType::Bishop:
101 if (adf != adr || adf == 0) return false;
102 return rayClear((df > 0) - (df < 0), (dr > 0) - (dr < 0));
103 case PieceType::Rook:
104 if (df != 0 && dr != 0) return false;
105 if (adf + adr == 0) return false;
106 return rayClear((df > 0) - (df < 0), (dr > 0) - (dr < 0));
107 case PieceType::Queen:
108 if (adf == adr && adf > 0)
109 return rayClear((df > 0) - (df < 0), (dr > 0) - (dr < 0));
110 if ((df == 0 || dr == 0) && (adf + adr > 0))
111 return rayClear((df > 0) - (df < 0), (dr > 0) - (dr < 0));
112 return false;
113 default:
114 return false;
115 }
116}

118bool ChessGame::resolveMove(Move& m, Color toMove) const {
119 if (m.isCastleKingside) {
120 int rank = (toMove == Color::White) ? 0 : 7;
121 m.from = {4, rank};
122 m.to = {6, rank};
123 return true;
124 }
125 if (m.isCastleQueenside) {
126 int rank = (toMove == Color::White) ? 0 : 7;
127 m.from = {4, rank};
128 m.to = {2, rank};
129 return true;
130 }
131
132 // Collect candidate pieces of matching type+color that can reach the destination.
133 std::vector<ChessPiece*> candidates;
134 for (auto& up : pieces_) {
135 ChessPiece* p = up.get();
136 if (p->captured()) continue;
137 if (p->color() != toMove) continue;
138 if (p->type() != m.piece) continue;
139 if (m.from.file >= 0 && p->square().file != m.from.file) continue;
140 if (m.from.rank >= 0 && p->square().rank != m.from.rank) continue;
141 if (!canPieceReach(*p, m.to, m.isCapture)) continue;
142 candidates.push_back(p);
143 }
144
145 if (candidates.empty()) {
146 // En passant fallback: pawn capture where target is empty.
147 if (m.piece == PieceType::Pawn && m.isCapture) {
148 for (auto& up : pieces_) {
149 ChessPiece* p = up.get();
150 if (p->captured() || p->color() != toMove || p->type() != PieceType::Pawn) continue;
151 if (m.from.file >= 0 && p->square().file != m.from.file) continue;
152 int dir = (toMove == Color::White) ? 1 : -1;
153 if (p->square().rank + dir != m.to.rank) continue;
154 if (std::abs(p->square().file - m.to.file) != 1) continue;
155 candidates.push_back(p);
156 }
157 }
158 }
159
160 if (candidates.empty()) return false;
161 m.from = candidates.front()->square();
162 return true;
163}

165void ChessGame::applyMoveLogical(const Move& m) {
166 if (m.isCastleKingside || m.isCastleQueenside) {
167 int rank = m.from.rank;
168 int rookFromFile = m.isCastleKingside ? 7 : 0;
169 int rookToFile = m.isCastleKingside ? 5 : 3;
170
171 ChessPiece* king = pieceAt(m.from);
172 ChessPiece* rook = pieceAt({rookFromFile, rank});
173 if (king) {
174 removeAt(m.from);
175 placePiece(king, m.to);
176 }
177 if (rook) {
178 removeAt({rookFromFile, rank});
179 placePiece(rook, {rookToFile, rank});
180 }
181 return;
182 }
183
184 ChessPiece* mover = pieceAt(m.from);
185 if (!mover) return;
186
187 // Capture
188 ChessPiece* target = pieceAt(m.to);
189 if (m.isCapture && !target && mover->type() == PieceType::Pawn) {
190 // en passant - captured pawn sits on (m.to.file, m.from.rank)
191 Square epSq{ m.to.file, m.from.rank };
192 if (auto* ep = pieceAt(epSq)) {
193 ep->setCaptured(true);
194 removeAt(epSq);
195 }
196 }
197 if (target) {
198 target->setCaptured(true);
199 removeAt(m.to);
200 }
201
202 removeAt(m.from);
203 placePiece(mover, m.to);
204
205 // Promotion
206 if (m.promotion != PieceType::None) {
207 mover->setType(m.promotion);
208 }
209}

211void ChessGame::applyMoveAnimated(const Move& m) {
212 const float duration = std::max(0.15f, 1.0f / speed_ * 0.7f);
213
214 if (m.isCastleKingside || m.isCastleQueenside) {
215 int rank = m.from.rank;
216 int rookFromFile = m.isCastleKingside ? 7 : 0;
217 int rookToFile = m.isCastleKingside ? 5 : 3;
218
219 if (auto* king = pieceAt(m.from)) {
220 king->animator().moveTo(squareToWorld(m.to), duration);
221 }
222 if (auto* rook = pieceAt({rookFromFile, rank})) {
223 rook->animator().moveTo(squareToWorld({rookToFile, rank}), duration);
224 }
225 } else {
226 ChessPiece* mover = pieceAt(m.from);
227 if (mover) {
228 // Tiny "hop" for knights (and slightly for captures).
229 float hop = 0.0f;
230 if (mover->type() == PieceType::Knight) hop = 0.45f;
231 else if (m.isCapture) hop = 0.15f;
232 mover->animator().setHopAmplitude(hop);
233 mover->animator().moveTo(squareToWorld(m.to), duration);
234 }
235 }
236 applyMoveLogical(m);
237}

239bool ChessGame::stepForward() {
240 if (currentMoveIndex_ >= static_cast<int>(moves_.size())) return false;
241 Move m = moves_[currentMoveIndex_];
242 Color toMove = (currentMoveIndex_ % 2 == 0) ? Color::White : Color::Black;
243 if (!resolveMove(m, toMove)) {
244 std::cerr << "[Chess3D] Failed to resolve move " << currentMoveIndex_ + 1
245 << ": " << m.san << '\n';
246 return false;
247 }
248 moves_[currentMoveIndex_] = m; // store resolved version (with from set)
249 applyMoveAnimated(m);
250 ++currentMoveIndex_;
251 return true;
252}

254bool ChessGame::stepBackward() {
255 if (currentMoveIndex_ <= 0) return false;
256 int target = currentMoveIndex_ - 1;
257
258 // Rebuild from scratch up to target moves applied, then snap visuals.
259 resetToStartingPosition();
260 for (int i = 0; i < target; ++i) {
261 Move m = moves_[i];
262 Color toMove = (i % 2 == 0) ? Color::White : Color::Black;
263 if (!resolveMove(m, toMove)) {
264 std::cerr << "[Chess3D] Rebuild failed at move " << i + 1 << '\n';
265 break;
266 }
267 moves_[i] = m;
268 applyMoveLogical(m);
269 }
270 for (auto& p : pieces_) {
271 if (!p->captured()) p->snapAnimatorToSquare();
272 }
273 currentMoveIndex_ = target;
274 return true;
275}

277void ChessGame::update(float dt) {
278 for (auto& p : pieces_) p->animator().update(dt);
279
280 if (playing_) {
281 autoPlayAccum_ += dt * speed_;
282 const float threshold = 1.0f;
283 if (autoPlayAccum_ >= threshold) {
284 autoPlayAccum_ -= threshold;
285 if (!stepForward()) {
286 playing_ = false;
287 autoPlayAccum_ = 0.0f;
288 }
289 }
290 } else {
291 autoPlayAccum_ = 0.0f;
292 }
293}

295bool ChessGame::loadPGNFile(const std::string& path) {
296 auto pr = PGNParser::parseFile(path);
297 if (!pr.error.empty()) {
298 std::cerr << "[Chess3D] PGN error: " << pr.error << '\n';
299 return false;
300 }
301 moves_ = std::move(pr.moves);
302 headers_ = std::move(pr.headers);
303 resetToStartingPosition();
304 for (auto& p : pieces_) p->snapAnimatorToSquare();
305 return true;
306}
