52bool Texture::uploadRGBA(const unsigned char* data, int w, int h, bool srgb, bool generateMips) {
53 if (texId_) { glDeleteTextures(1, &texId_); texId_ = 0; }
54 glGenTextures(1, &texId_);
55 glBindTexture(GL_TEXTURE_2D, texId_);
56
57 GLenum internalFmt = srgb ? GL_SRGB8_ALPHA8 : GL_RGBA8;
58 glTexImage2D(GL_TEXTURE_2D, 0, internalFmt, w, h, 0, GL_RGBA, GL_UNSIGNED_BYTE, data);
59
60 if (generateMips) glGenerateMipmap(GL_TEXTURE_2D);
61
62 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
63 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
64 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER,
65 generateMips ? GL_LINEAR_MIPMAP_LINEAR : GL_LINEAR);
66 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
67
68 GLfloat maxAniso = 1.0f;
69 glGetFloatv(GL_MAX_TEXTURE_MAX_ANISOTROPY_EXT, &maxAniso);
70 if (maxAniso > 1.0f) {
71 glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT, std::min(maxAniso, 8.0f));
72 }
73
74 glBindTexture(GL_TEXTURE_2D, 0);
75
76 width_ = w;
77 height_ = h;
78 return true;
79}

81bool Texture::loadFromFile(const std::string& path, bool srgb, bool generateMips) {
82 int w = 0, h = 0, c = 0;
83 stbi_set_flip_vertically_on_load(true);
84 unsigned char* data = stbi_load(path.c_str(), &w, &h, &c, 4);
85 if (!data) {
86 std::cerr << "[Texture] cannot load \"" << path << "\": " << stbi_failure_reason() << '\n';
87 return false;
88 }
89 bool ok = uploadRGBA(data, w, h, srgb, generateMips);
90 stbi_image_free(data);
91 return ok;
92}

97float hash2(int x, int y, std::uint32_t seed) {
98 std::uint32_t h = static_cast<std::uint32_t>(x) * 374761393u
99 + static_cast<std::uint32_t>(y) * 668265263u
100 + seed;
101 h = (h ^ (h >> 13)) * 1274126177u;
102 return static_cast<float>(h & 0x00FFFFFFu) / static_cast<float>(0x01000000);
103}

105float smoothNoise(float x, float y, std::uint32_t seed) {
106 int x0 = static_cast<int>(std::floor(x));
107 int y0 = static_cast<int>(std::floor(y));
108 float tx = x - x0;
109 float ty = y - y0;
110 auto fade = [](float t){ return t*t*(3.0f - 2.0f*t); };
111 float u = fade(tx);
112 float v = fade(ty);
113 float a = hash2(x0, y0, seed);
114 float b = hash2(x0+1, y0, seed);
115 float c = hash2(x0, y0+1, seed);
116 float d = hash2(x0+1, y0+1, seed);
117 float ab = a + (b - a) * u;
118 float cd = c + (d - c) * u;
119 return ab + (cd - ab) * v;
120}

122float fbm(float x, float y, std::uint32_t seed, int octaves = 4) {
123 float amp = 0.5f, freq = 1.0f, sum = 0.0f, norm = 0.0f;
124 for (int i = 0; i < octaves; ++i) {
125 sum += amp * smoothNoise(x * freq, y * freq, seed + i*131);
126 norm += amp;
127 amp *= 0.5f;
128 freq *= 2.0f;
129 }
130 return sum / norm;
131}

135bool Texture::createProceduralWood(int size, bool lightWood) {
136 std::vector<unsigned char> pixels(static_cast<std::size_t>(size) * size * 4);
137
138 glm::vec3 base, dark;
139 std::uint32_t seed;
140 if (lightWood) {
141 base = {0.85f, 0.70f, 0.46f}; // maple
142 dark = {0.55f, 0.38f, 0.20f};
143 seed = 1337u;
144 } else {
145 base = {0.32f, 0.20f, 0.12f}; // walnut
146 dark = {0.10f, 0.06f, 0.04f};
147 seed = 9001u;
148 }
149
150 for (int y = 0; y < size; ++y) {
151 for (int x = 0; x < size; ++x) {
152 float u = static_cast<float>(x) / size;
153 float v = static_cast<float>(y) / size;
154
155 // Tight ring pattern + low-freq fbm warp -> wood grain.
156 float warp = fbm(u * 6.0f, v * 6.0f, seed, 4) * 0.4f;
157 float rings = std::sin((v * 22.0f + warp * 14.0f) * 3.14159f);
158 float grain = fbm(u * 64.0f, v * 12.0f, seed + 5, 3);
159
160 float t = 0.5f + 0.5f * rings;
161 t = glm::clamp(t * 0.8f + grain * 0.4f, 0.0f, 1.0f);
162
163 glm::vec3 col = glm::mix(dark, base, t);
164
165 int idx = (y * size + x) * 4;
166 pixels[idx + 0] = static_cast<unsigned char>(glm::clamp(col.r, 0.0f, 1.0f) * 255.0f);
167 pixels[idx + 1] = static_cast<unsigned char>(glm::clamp(col.g, 0.0f, 1.0f) * 255.0f);
168 pixels[idx + 2] = static_cast<unsigned char>(glm::clamp(col.b, 0.0f, 1.0f) * 255.0f);
169 pixels[idx + 3] = 255;
170 }
171 }
172 return uploadRGBA(pixels.data(), size, size, /*srgb*/true, /*mips*/true);
173}

175bool Texture::createProceduralMarble(int size, bool whiteMarble) {
176 std::vector<unsigned char> pixels(static_cast<std::size_t>(size) * size * 4);
177
178 glm::vec3 base, vein;
179 std::uint32_t seed;
180 if (whiteMarble) {
181 base = {0.93f, 0.92f, 0.88f};
182 vein = {0.55f, 0.55f, 0.55f};
183 seed = 4242u;
184 } else {
185 base = {0.12f, 0.10f, 0.10f};
186 vein = {0.35f, 0.30f, 0.28f};
187 seed = 7777u;
188 }
189
190 for (int y = 0; y < size; ++y) {
191 for (int x = 0; x < size; ++x) {
192 float u = static_cast<float>(x) / size;
193 float v = static_cast<float>(y) / size;
194
195 float n = fbm(u * 5.0f, v * 5.0f, seed, 5);
196 float n2 = fbm(u * 14.0f + 3.0f, v * 14.0f + 7.0f, seed + 1, 4);
197 float veins = std::abs(std::sin((u + v + n2) * 11.0f + n * 6.0f));
198 veins = std::pow(1.0f - veins, 6.0f);
199
200 glm::vec3 col = glm::mix(base, vein, veins);
201
202 int idx = (y * size + x) * 4;
203 pixels[idx + 0] = static_cast<unsigned char>(glm::clamp(col.r, 0.0f, 1.0f) * 255.0f);
204 pixels[idx + 1] = static_cast<unsigned char>(glm::clamp(col.g, 0.0f, 1.0f) * 255.0f);
205 pixels[idx + 2] = static_cast<unsigned char>(glm::clamp(col.b, 0.0f, 1.0f) * 255.0f);
206 pixels[idx + 3] = 255;
207 }
208 }
209 return uploadRGBA(pixels.data(), size, size, /*srgb*/true, /*mips*/true);
210}
