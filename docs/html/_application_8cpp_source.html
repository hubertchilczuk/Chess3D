53bool Application::initialize(int width, int height, const std::string& title) {
54 glfwSetErrorCallback(glfwErrorCallback);
55 if (!glfwInit()) {
56 std::cerr << "[Chess3D] glfwInit failed\n";
57 return false;
58 }
59
60 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
61 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
62 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
63 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GLFW_TRUE);
64 glfwWindowHint(GLFW_SAMPLES, 4);
65
66 window_ = glfwCreateWindow(width, height, title.c_str(), nullptr, nullptr);
67 if (!window_) {
68 std::cerr << "[Chess3D] glfwCreateWindow failed\n";
69 glfwTerminate();
70 return false;
71 }
72 width_ = width;
73 height_ = height;
74
75 glfwMakeContextCurrent(window_);
76 glfwSwapInterval(1);
77 glfwSetWindowUserPointer(window_, this);
78
79 glfwSetFramebufferSizeCallback(window_, cbResize);
80 glfwSetMouseButtonCallback (window_, cbMouseButton);
81 glfwSetCursorPosCallback (window_, cbCursorPos);
82 glfwSetScrollCallback (window_, cbScroll);
83 glfwSetKeyCallback (window_, cbKey);
84
85 glewExperimental = GL_TRUE;
86 if (GLenum err = glewInit(); err != GLEW_OK) {
87 std::cerr << "[Chess3D] glewInit failed: "
88 << reinterpret_cast<const char*>(glewGetErrorString(err)) << '\n';
89 return false;
90 }
91 glGetError(); // GLEW sometimes leaves a benign INVALID_ENUM on init.
92 glEnable(GL_MULTISAMPLE);
93
94 std::cout << "[Chess3D] OpenGL " << glGetString(GL_VERSION)
95 << " | Renderer: " << glGetString(GL_RENDERER) << '\n';
96
97 renderer_ = std::make_unique<Renderer>();
98 if (!renderer_->initialize()) {
99 std::cerr << "[Chess3D] renderer initialization failed\n";
100 return false;
101 }
102 int fbW, fbH;
103 glfwGetFramebufferSize(window_, &fbW, &fbH);
104 renderer_->resize(fbW, fbH);
105
106 camera_ = std::make_unique<Camera>(glm::vec3(0.0f, 7.5f, 9.0f), -90.0f, -42.0f);
107 camera_->lookAt(glm::vec3(0.0f, 0.0f, 0.0f));
108
109 game_ = std::make_unique<ChessGame>();
110 {
111 std::string defaultPgn = std::string(CHESS3D_RESOURCES_DIR) + "/pgn/sample_game.pgn";
112 if (std::filesystem::exists(defaultPgn)) {
113 game_->loadPGNFile(defaultPgn);
114 }
115 }
116
117 ui_ = std::make_unique<UIManager>();
118 if (!ui_->initialize(window_)) {
119 std::cerr << "[Chess3D] UI manager init failed\n";
120 return false;
121 }
122
123 lastFrameTime_ = static_cast<float>(glfwGetTime());
124 return true;
125}

145void Application::onMouseButton(int button, int action) {
146 if (ui_ && ui_->wantCaptureMouse()) return;
147 if (button == GLFW_MOUSE_BUTTON_RIGHT) {
148 rightMouseDown_ = (action == GLFW_PRESS);
149 if (rightMouseDown_) {
150 firstMouse_ = true;
151 glfwSetInputMode(window_, GLFW_CURSOR, GLFW_CURSOR_DISABLED);
152 } else {
153 glfwSetInputMode(window_, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
154 }
155 }

158void Application::onMouseMove(double xpos, double ypos) {
159 if (firstMouse_) {
160 lastMouseX_ = xpos;
161 lastMouseY_ = ypos;
162 firstMouse_ = false;
163 return;
164 }
165 float dx = static_cast<float>(xpos - lastMouseX_);
166 float dy = static_cast<float>(ypos - lastMouseY_);
167 lastMouseX_ = xpos;
168 lastMouseY_ = ypos;
169
170 if (rightMouseDown_ && camera_) camera_->processMouse(dx, dy);
171}

178void Application::onKey(int key, int action) {
179 if (ui_ && ui_->wantCaptureKeyboard()) return;
180 if (action != GLFW_PRESS) return;
181 switch (key) {
182 case GLFW_KEY_ESCAPE: glfwSetWindowShouldClose(window_, GLFW_TRUE); break;
183 case GLFW_KEY_SPACE: if (game_) game_->togglePlay(); break;
184 case GLFW_KEY_RIGHT: if (game_) game_->stepForward(); break;
185 case GLFW_KEY_LEFT: if (game_) game_->stepBackward(); break;
186 case GLFW_KEY_R: if (game_) game_->resetToStartingPosition(); break;
187 default: break;
188 }
189}

191void Application::processInput(float dt) {
192 if (!camera_) return;
193 if (ui_ && ui_->wantCaptureKeyboard()) return;
194
195 glm::vec3 dir(0.0f);
196 if (glfwGetKey(window_, GLFW_KEY_W) == GLFW_PRESS) dir.z += 1.0f;
197 if (glfwGetKey(window_, GLFW_KEY_S) == GLFW_PRESS) dir.z -= 1.0f;
198 if (glfwGetKey(window_, GLFW_KEY_D) == GLFW_PRESS) dir.x += 1.0f;
199 if (glfwGetKey(window_, GLFW_KEY_A) == GLFW_PRESS) dir.x -= 1.0f;
200 if (glfwGetKey(window_, GLFW_KEY_E) == GLFW_PRESS) dir.y += 1.0f;
201 if (glfwGetKey(window_, GLFW_KEY_Q) == GLFW_PRESS) dir.y -= 1.0f;
202
203 if (glm::length(dir) > 0.0f) camera_->processKeyboard(dir, dt);
204}

211void Application::render() {
212 renderer_->render(*camera_, *game_, static_cast<float>(glfwGetTime()));
213
214 ui_->beginFrame();
215 ui_->draw(*game_, *renderer_, *camera_, lastFps_);
216 ui_->endFrame();
217
218 glfwSwapBuffers(window_);
219}

221void Application::run() {
222 while (!glfwWindowShouldClose(window_)) {
223 float now = static_cast<float>(glfwGetTime());
224 float dt = now - lastFrameTime_;
225 if (dt > 0.1f) dt = 0.1f; // cap large frame steps
226 lastFrameTime_ = now;
227
228 // FPS accumulator
229 fpsAccum_ += dt;
230 fpsFrames_ += 1;
231 if (fpsAccum_ >= 0.5f) {
232 lastFps_ = static_cast<float>(fpsFrames_) / fpsAccum_;
233 fpsAccum_ = 0.0f;
234 fpsFrames_ = 0;
235 }
236
237 glfwPollEvents();
238 processInput(dt);
239 update(dt);
240 render();
241 }
