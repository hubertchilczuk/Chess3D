20void Model::normalize() {
21 if (!boundsInit_) {
22 normalizationTransform_ = glm::mat4(1.0f);
23 return;
24 }
25
26 // Step 1: pick an axis fix. Chess pieces are tall, so whichever axis has
27 // the largest extent in the imported model is the "up" axis. We rotate it
28 // to match our engine's +Y up convention. Files exported as Y-up (most
29 // .obj/.fbx) need no rotation; .blend / 3ds Max files with Z-up are
30 // detected automatically.
31 const glm::vec3 ext = boundsMax_ - boundsMin_;
32 glm::mat4 axisFix(1.0f);
33 if (ext.z > ext.y && ext.z >= ext.x) {
34 // Z is the tallest axis -> rotate Z-up to Y-up.
35 axisFix = glm::rotate(glm::mat4(1.0f), -glm::half_pi<float>(), glm::vec3(1, 0, 0));
36 } else if (ext.x > ext.y && ext.x > ext.z) {
37 // X is the tallest axis (rare) -> rotate to Y-up.
38 axisFix = glm::rotate(glm::mat4(1.0f), glm::half_pi<float>(), glm::vec3(0, 0, 1));
39 }
40 // else: Y already up, no rotation needed.
41
42 // Step 2: compute axis-fixed bounding box by transforming the 8 corners.
43 const glm::vec3 corners[8] = {
44 {boundsMin_.x, boundsMin_.y, boundsMin_.z},
45 {boundsMax_.x, boundsMin_.y, boundsMin_.z},
46 {boundsMin_.x, boundsMax_.y, boundsMin_.z},
47 {boundsMax_.x, boundsMax_.y, boundsMin_.z},
48 {boundsMin_.x, boundsMin_.y, boundsMax_.z},
49 {boundsMax_.x, boundsMin_.y, boundsMax_.z},
50 {boundsMin_.x, boundsMax_.y, boundsMax_.z},
51 {boundsMax_.x, boundsMax_.y, boundsMax_.z},
52 };
53 glm::vec3 rmin( std::numeric_limits<float>::max());
54 glm::vec3 rmax(-std::numeric_limits<float>::max());
55 for (const auto& c : corners) {
56 glm::vec3 r = glm::vec3(axisFix * glm::vec4(c, 1.0f));
57 rmin = glm::min(rmin, r);
58 rmax = glm::max(rmax, r);
59 }
60 glm::vec3 center = (rmin + rmax) * 0.5f;
61 float height = std::max(rmax.y - rmin.y, 0.0001f);
62
63 // Step 3: target height matches a typical chess piece visually.
64 // Pawn-sized scenes will be small, king-sized scenes will be tall;
65 // a single global per-model scale keeps relative proportions intact.
66 constexpr float kTargetHeight = 1.4f;
67 const float scale = kTargetHeight / height;
68
69 // Step 4: place the centre on the XZ axes and drop the base to y = 0
70 // (translation applied AFTER the axis rotation).
71 const glm::vec3 translation(-center.x, -rmin.y, -center.z);
72
73 normalizationTransform_ =
74 glm::scale (glm::mat4(1.0f), glm::vec3(scale)) *
75 glm::translate(glm::mat4(1.0f), translation) *
76 axisFix;
77}

88// Tiny union-find over vertex indices.
89struct UnionFind {
90 std::vector<int> parent;
91 explicit UnionFind(std::size_t n) : parent(n) {
92 for (std::size_t i = 0; i < n; ++i) parent[i] = static_cast<int>(i);
93 }
94 int find(int x) {
95 while (parent[x] != x) {
96 parent[x] = parent[parent[x]];
97 x = parent[x];
98 }
99 return x;
100 }
101 void unite(int a, int b) {
102 a = find(a); b = find(b);
103 if (a != b) parent[b] = a;
104 }
105};

109std::vector<std::unique_ptr<Model>>
110Model::loadAndSplitByComponents(const std::string& path) {
111 Assimp::Importer importer;
112 const aiScene* scene = importer.ReadFile(path,
113 aiProcess_Triangulate |
114 aiProcess_GenSmoothNormals |
115 aiProcess_JoinIdenticalVertices |
116 aiProcess_ImproveCacheLocality);
117 if (!scene || scene->mFlags & AI_SCENE_FLAGS_INCOMPLETE || !scene->mRootNode) {
118 std::cerr << "[Model] Assimp error for \"" << path << "\": "
119 << importer.GetErrorString() << '\n';
120 return {};
121 }
122
123 // 1) Flatten the scene into one big vertex+index buffer. We can ignore
124 // node transforms here because chess-piece meshes typically come in
125 // a single shared frame already.
126 std::vector<Vertex> allVerts;
127 std::vector<std::uint32_t> allIdx;
128 for (unsigned mi = 0; mi < scene->mNumMeshes; ++mi) {
129 const aiMesh* mesh = scene->mMeshes[mi];
130 const std::uint32_t base = static_cast<std::uint32_t>(allVerts.size());
131
132 for (unsigned i = 0; i < mesh->mNumVertices; ++i) {
133 Vertex v;
134 v.position = { mesh->mVertices[i].x, mesh->mVertices[i].y, mesh->mVertices[i].z };
135 v.normal = mesh->HasNormals()
136 ? glm::vec3(mesh->mNormals[i].x, mesh->mNormals[i].y, mesh->mNormals[i].z)
137 : glm::vec3(0.0f, 1.0f, 0.0f);
138 v.uv = mesh->HasTextureCoords(0)
139 ? glm::vec2(mesh->mTextureCoords[0][i].x, mesh->mTextureCoords[0][i].y)
140 : glm::vec2(v.position.x * 0.5f + 0.5f, v.position.z * 0.5f + 0.5f);
141 allVerts.push_back(v);
142 }
143 for (unsigned f = 0; f < mesh->mNumFaces; ++f) {
144 const aiFace& face = mesh->mFaces[f];
145 if (face.mNumIndices != 3) continue;
146 allIdx.push_back(base + face.mIndices[0]);
147 allIdx.push_back(base + face.mIndices[1]);
148 allIdx.push_back(base + face.mIndices[2]);
149 }
150 }
151 if (allVerts.empty() || allIdx.size() < 3) return {};
152
153 // 2) Union-find: join every pair of indices that appear together in a
154 // triangle. Connected components in the index space correspond to
155 // spatially disjoint sub-meshes (i.e. separate chess pieces).
156 UnionFind uf(allVerts.size());
157 for (std::size_t i = 0; i + 2 < allIdx.size(); i += 3) {
158 uf.unite(static_cast<int>(allIdx[i]), static_cast<int>(allIdx[i + 1]));
159 uf.unite(static_cast<int>(allIdx[i + 1]), static_cast<int>(allIdx[i + 2]));
160 }
161
162 // 3) Bucket faces by their root component.
163 std::unordered_map<int, std::vector<std::uint32_t>> facesByRoot;
164 facesByRoot.reserve(16);
165 for (std::size_t i = 0; i + 2 < allIdx.size(); i += 3) {
166 int root = uf.find(static_cast<int>(allIdx[i]));
167 auto& bucket = facesByRoot[root];
168 bucket.push_back(allIdx[i]);
169 bucket.push_back(allIdx[i + 1]);
170 bucket.push_back(allIdx[i + 2]);
171 }
172
173 // 4) Build a Model per component, reindexing vertices densely.
174 struct BuiltComponent {
175 std::unique_ptr<Model> model;
176 float height = 0.0f;
177 };
178 std::vector<BuiltComponent> built;
179 built.reserve(facesByRoot.size());
180
181 for (auto& kv : facesByRoot) {
182 std::vector<Vertex> cVerts;
183 std::vector<std::uint32_t> cIdx;
184 std::unordered_map<std::uint32_t, std::uint32_t> remap;
185 cVerts.reserve(kv.second.size());
186 cIdx.reserve(kv.second.size());
187
188 glm::vec3 bmin( std::numeric_limits<float>::max());
189 glm::vec3 bmax(-std::numeric_limits<float>::max());
190
191 for (std::uint32_t oldI : kv.second) {
192 auto it = remap.find(oldI);
193 std::uint32_t newI;
194 if (it == remap.end()) {
195 newI = static_cast<std::uint32_t>(cVerts.size());
196 cVerts.push_back(allVerts[oldI]);
197 remap[oldI] = newI;
198 bmin = glm::min(bmin, allVerts[oldI].position);
199 bmax = glm::max(bmax, allVerts[oldI].position);
200 } else {
201 newI = it->second;
202 }
203 cIdx.push_back(newI);
204 }
205
206 if (cVerts.empty() || cIdx.size() < 3) continue;
207
208 auto model = std::make_unique<Model>();
209 auto mesh = std::make_unique<Mesh>();
210 mesh->upload(cVerts, cIdx);
211 model->meshes_.push_back(std::move(mesh));
212 model->boundsMin_ = bmin;
213 model->boundsMax_ = bmax;
214 model->boundsInit_ = true;
215 model->normalize();
216
217 BuiltComponent bc;
218 bc.height = bmax.y - bmin.y;
219 bc.model = std::move(model);
220 built.push_back(std::move(bc));
221 }
222
223 // 5) Drop tiny floaters (e.g. helper geometry left in the scene) - keep
224 // components whose face count is at least a small fraction of the median.
225 if (built.size() > 6) {
226 std::sort(built.begin(), built.end(),
227 [](const BuiltComponent& a, const BuiltComponent& b) {
228 return a.model->meshes_[0]->indexCount() >
229 b.model->meshes_[0]->indexCount();
230 });
231 built.resize(6);
232 }
233
234 // 6) Final sort by height ascending so the renderer can map components
235 // to pieces by their typical relative heights.
236 std::sort(built.begin(), built.end(),
237 [](const BuiltComponent& a, const BuiltComponent& b) {
238 return a.height < b.height;
239 });
240
241 std::vector<std::unique_ptr<Model>> out;
242 out.reserve(built.size());
243 for (auto& bc : built) out.push_back(std::move(bc.model));
244 return out;
245}
