38static std::string slurp(const std::string& path) {
39 std::ifstream f(path, std::ios::binary);
40 if (!f.is_open()) return {};
41 std::ostringstream ss;
42 ss << f.rdbuf();
43 return ss.str();
44}

46GLuint Shader::compileStage(GLenum stage, const std::string& source, const std::string& tag) {
47 GLuint sh = glCreateShader(stage);
48 const char* src = source.c_str();
49 glShaderSource(sh, 1, &src, nullptr);
50 glCompileShader(sh);
51
52 GLint ok = 0;
53 glGetShaderiv(sh, GL_COMPILE_STATUS, &ok);
54 if (!ok) {
55 GLint len = 0;
56 glGetShaderiv(sh, GL_INFO_LOG_LENGTH, &len);
57 std::string log(static_cast<std::size_t>(std::max(len, 1)), '\0');
58 glGetShaderInfoLog(sh, len, nullptr, log.data());
59 std::cerr << "[Shader] " << tag << " compile error:\n" << log << '\n';
60 glDeleteShader(sh);
61 return 0;
62 }
63 return sh;
64}

66bool Shader::loadFromSources(const std::string& vsSrc, const std::string& fsSrc) {
67 GLuint vs = compileStage(GL_VERTEX_SHADER, vsSrc, "vertex");
68 GLuint fs = compileStage(GL_FRAGMENT_SHADER, fsSrc, "fragment");
69 if (!vs || !fs) {
70 if (vs) glDeleteShader(vs);
71 if (fs) glDeleteShader(fs);
72 return false;
73 }
74
75 GLuint prog = glCreateProgram();
76 glAttachShader(prog, vs);
77 glAttachShader(prog, fs);
78 glLinkProgram (prog);
79
80 GLint ok = 0;
81 glGetProgramiv(prog, GL_LINK_STATUS, &ok);
82 if (!ok) {
83 GLint len = 0;
84 glGetProgramiv(prog, GL_INFO_LOG_LENGTH, &len);
85 std::string log(static_cast<std::size_t>(std::max(len, 1)), '\0');
86 glGetProgramInfoLog(prog, len, nullptr, log.data());
87 std::cerr << "[Shader] link error:\n" << log << '\n';
88 glDeleteShader(vs);
89 glDeleteShader(fs);
90 glDeleteProgram(prog);
91 return false;
92 }
93
94 glDetachShader(prog, vs);
95 glDetachShader(prog, fs);
96 glDeleteShader(vs);
97 glDeleteShader(fs);
98
99 if (programId_) glDeleteProgram(programId_);
100 programId_ = prog;
101 uniformCache_.clear();
102 return true;
103}

105bool Shader::loadFromFiles(const std::string& vertexPath, const std::string& fragmentPath) {
106 std::string vs = slurp(vertexPath);
107 std::string fs = slurp(fragmentPath);
108 if (vs.empty()) {
109 std::cerr << "[Shader] cannot read vertex shader: " << vertexPath << '\n';
110 return false;
111 }
112 if (fs.empty()) {
113 std::cerr << "[Shader] cannot read fragment shader: " << fragmentPath << '\n';
114 return false;
115 }
116 return loadFromSources(vs, fs);
117}

121GLint Shader::uniformLocation(const std::string& name) const {
122 auto it = uniformCache_.find(name);
123 if (it != uniformCache_.end()) return it->second;
124 GLint loc = glGetUniformLocation(programId_, name.c_str());
125 uniformCache_.emplace(name, loc);
