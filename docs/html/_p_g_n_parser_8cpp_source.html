38std::string PGNParser::stripComments(const std::string& text) {
39 std::string out;
40 out.reserve(text.size());
41 int braceDepth = 0;
42 int parenDepth = 0;
43 bool lineComment = false;
44
45 for (char c : text) {
46 if (lineComment) {
47 if (c == '\n' || c == '\r') lineComment = false;
48 continue;
49 }
50 if (c == ';') { lineComment = true; continue; }
51 if (c == '{') { ++braceDepth; continue; }
52 if (c == '}') { if (braceDepth) --braceDepth; continue; }
53 if (c == '(') { ++parenDepth; continue; }
54 if (c == ')') { if (parenDepth) --parenDepth; continue; }
55 if (braceDepth || parenDepth) continue;
56 out.push_back(c);
57 }
58 return out;
59}

61bool PGNParser::parseSAN(const std::string& sanIn, Color toMove, Move& out) {
62 out = Move{};
63 out.color = toMove;
64
65 std::string s = sanIn;
66
67 // Strip annotation suffixes (!, ?, !!, ??, !?, ?!)
68 while (!s.empty() && (s.back() == '!' || s.back() == '?')) s.pop_back();
69 if (s.empty()) return false;
70
71 // Check / mate suffix
72 if (s.back() == '#') { out.isMate = true; s.pop_back(); }
73 else if (s.back() == '+') { out.isCheck = true; s.pop_back(); }
74 if (s.empty()) return false;
75
76 // Castling
77 if (s == "O-O" || s == "0-0") {
78 out.isCastleKingside = true;
79 out.piece = PieceType::King;
80 out.san = sanIn;
81 return true;
82 }
83 if (s == "O-O-O" || s == "0-0-0") {
84 out.isCastleQueenside = true;
85 out.piece = PieceType::King;
86 out.san = sanIn;
87 return true;
88 }
89
90 // Promotion ("=Q" or just trailing letter on some old PGNs)
91 auto eq = s.find('=');
92 if (eq != std::string::npos && eq + 1 < s.size()) {
93 out.promotion = pieceFromLetter(s[eq + 1]);
94 s.erase(eq);
95 } else if (s.size() >= 3) {
96 // tolerate "e8Q" - last char uppercase piece letter, second-to-last digit
97 char c1 = s[s.size() - 1];
98 char c2 = s[s.size() - 2];
99 if (std::isupper(static_cast<unsigned char>(c1)) && std::isdigit(static_cast<unsigned char>(c2))
100 && (c1 == 'Q' || c1 == 'R' || c1 == 'B' || c1 == 'N')) {
101 out.promotion = pieceFromLetter(c1);
102 s.pop_back();
103 }
104 }
105
106 if (s.size() < 2) return false;
107
108 // Destination = last 2 chars [a-h][1-8]
109 char destR = s.back(); s.pop_back();
110 char destF = s.back(); s.pop_back();
111 if (destF < 'a' || destF > 'h' || destR < '1' || destR > '8') return false;
112 out.to = { destF - 'a', destR - '1' };
113
114 // Capture marker
115 if (!s.empty() && s.back() == 'x') {
116 out.isCapture = true;
117 s.pop_back();
118 }
119
120 // Piece type (first remaining char if uppercase)
121 if (!s.empty() && std::isupper(static_cast<unsigned char>(s.front()))) {
122 out.piece = pieceFromLetter(s.front());
123 s.erase(0, 1);
124 } else {
125 out.piece = PieceType::Pawn;
126 }
127
128 // Remaining chars (0..2) are disambiguation hints.
129 for (char c : s) {
130 if (c >= 'a' && c <= 'h') out.from.file = c - 'a';
131 else if (c >= '1' && c <= '8') out.from.rank = c - '1';
132 // ignore anything else silently
133 }
134
135 out.san = sanIn;
136 return true;
137}

139PGNParser::ParseResult PGNParser::parseString(const std::string& text) {
140 ParseResult result;
141
142 // Split headers and movetext.
143 std::string moveText;
144 {
145 std::istringstream iss(text);
146 std::string line;
147 while (std::getline(iss, line)) {
148 std::string t = trim(line);
149 if (t.empty()) { moveText += '\n'; continue; }
150 if (t.front() == '[' && t.back() == ']') {
151 // Header: [Key "Value"]
152 auto sp = t.find(' ');
153 auto q1 = t.find('"');
154 auto q2 = t.rfind('"');
155 if (sp != std::string::npos && q1 != std::string::npos && q2 > q1) {
156 result.headers.emplace_back(
157 t.substr(1, sp - 1),
158 t.substr(q1 + 1, q2 - q1 - 1));
159 }
160 } else {
161 moveText += t;
162 moveText += ' ';
163 }
164 }
165 }
166
167 moveText = stripComments(moveText);
168
169 std::istringstream toks(moveText);
170 std::string tok;
171 Color toMove = Color::White;
172
173 while (toks >> tok) {
174 if (tok.empty()) continue;
175 if (isResult(tok)) break;
176
177 // Skip NAGs ($1, $14, ...).
178 if (tok.front() == '$') continue;
179
180 // Strip leading move numbers like "1." or "12..."
181 if (std::isdigit(static_cast<unsigned char>(tok.front()))) {
182 auto dot = tok.find('.');
183 if (dot != std::string::npos) {
184 std::string rest = tok.substr(dot + 1);
185 while (!rest.empty() && rest.front() == '.') rest.erase(rest.begin());
186 if (rest.empty()) continue; // pure move number, e.g. "1."
187 tok = rest;
188 } else {
189 // Bare number? Skip.
190 continue;
191 }
192 }
193
194 if (tok.empty() || isResult(tok)) {
195 if (!tok.empty() && isResult(tok)) break;
196 continue;
197 }
198
199 Move m;
200 if (!parseSAN(tok, toMove, m)) {
201 result.error = "Cannot parse SAN token: \"" + tok + "\"";
202 return result;
203 }
204 result.moves.push_back(std::move(m));
205 toMove = opposite(toMove);
206 }
207
208 return result;
209}

211PGNParser::ParseResult PGNParser::parseFile(const std::string& path) {
212 std::ifstream f(path, std::ios::binary);
213 if (!f.is_open()) {
214 ParseResult r;
215 r.error = "Cannot open file: " + path;
216 return r;
217 }
218 std::ostringstream ss;
219 ss << f.rdbuf();
220 return parseString(ss.str());
221}
