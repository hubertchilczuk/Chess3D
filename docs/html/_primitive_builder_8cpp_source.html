18void appendBox(std::vector<Vertex>& v, std::vector<std::uint32_t>& idx,

21 const glm::vec3 p[8] = {
22 c + glm::vec3(-h.x, -h.y, -h.z),
23 c + glm::vec3( h.x, -h.y, -h.z),
24 c + glm::vec3( h.x, h.y, -h.z),
25 c + glm::vec3(-h.x, h.y, -h.z),
26 c + glm::vec3(-h.x, -h.y, h.z),
27 c + glm::vec3( h.x, -h.y, h.z),
28 c + glm::vec3( h.x, h.y, h.z),
29 c + glm::vec3(-h.x, h.y, h.z),
30 };

32 auto face = [&](int a, int b, int cIdx, int d, const glm::vec3& n) {
33 std::uint32_t i0 = pushVertex(v, p[a], n, {0,0});
34 std::uint32_t i1 = pushVertex(v, p[b], n, {1,0});
35 std::uint32_t i2 = pushVertex(v, p[cIdx], n, {1,1});
36 std::uint32_t i3 = pushVertex(v, p[d], n, {0,1});
37 idx.insert(idx.end(), { i0, i1, i2, i0, i2, i3 });
38 };
