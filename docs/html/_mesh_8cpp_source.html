41void Mesh::upload(const std::vector<Vertex>& vertices,
42 const std::vector<std::uint32_t>& indices) {
43 releaseGL();
44
45 glGenVertexArrays(1, &vao_);
46 glGenBuffers (1, &vbo_);
47 glGenBuffers (1, &ebo_);
48
49 glBindVertexArray(vao_);
50
51 glBindBuffer(GL_ARRAY_BUFFER, vbo_);
52 glBufferData(GL_ARRAY_BUFFER,
53 static_cast<GLsizeiptr>(vertices.size() * sizeof(Vertex)),
54 vertices.data(),
55 GL_STATIC_DRAW);
56
57 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ebo_);
58 glBufferData(GL_ELEMENT_ARRAY_BUFFER,
59 static_cast<GLsizeiptr>(indices.size() * sizeof(std::uint32_t)),
60 indices.data(),
61 GL_STATIC_DRAW);
62
63 // position
64 glEnableVertexAttribArray(0);
65 glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex),
66 reinterpret_cast<void*>(offsetof(Vertex, position)));
67 // normal
68 glEnableVertexAttribArray(1);
69 glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex),
70 reinterpret_cast<void*>(offsetof(Vertex, normal)));
71 // uv
72 glEnableVertexAttribArray(2);
73 glVertexAttribPointer(2, 2, GL_FLOAT, GL_FALSE, sizeof(Vertex),
74 reinterpret_cast<void*>(offsetof(Vertex, uv)));
75
76 glBindVertexArray(0);
77
78 indexCount_ = static_cast<GLsizei>(indices.size());
79}
