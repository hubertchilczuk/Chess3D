19void ChessBoard::build() {
20 // Each square -> its own little Mesh -> 0.5 unit cube of height 0.05.
21 const float thickness = 0.06f;
22
23 for (int r = 0; r < 8; ++r) {
24 for (int f = 0; f < 8; ++f) {
25 std::vector<Vertex> v;
26 std::vector<std::uint32_t> i;
27 glm::vec3 center = squareToWorld({f, r});
28 center.y = -thickness;
29 primitives::appendBox(v, i, center,
30 glm::vec3(kSquareSize * 0.5f, thickness, kSquareSize * 0.5f));
31
32 auto mesh = std::make_unique<Mesh>();
33 mesh->upload(v, i);
34 squares_[index(f, r)] = std::move(mesh);
35 }
36 }
37
38 // Decorative outer frame around the whole board.
39 {
40 std::vector<Vertex> v;
41 std::vector<std::uint32_t> i;
42 const float frameW = 0.6f;
43 const float frameH = 0.18f;
44 const float halfFull = kBoardHalf + frameW * 0.5f;
45 const float yC = -frameH * 0.5f;
46
47 // top (+Z) and bottom (-Z) strips
48 primitives::appendBox(v, i, {0, yC, kBoardHalf + frameW * 0.5f},
49 {halfFull, frameH * 0.5f, frameW * 0.5f});
50 primitives::appendBox(v, i, {0, yC, -kBoardHalf - frameW * 0.5f},
51 {halfFull, frameH * 0.5f, frameW * 0.5f});
52 // left (-X) and right (+X) strips
53 primitives::appendBox(v, i, { kBoardHalf + frameW * 0.5f, yC, 0},
54 {frameW * 0.5f, frameH * 0.5f, kBoardHalf});
55 primitives::appendBox(v, i, {-kBoardHalf - frameW * 0.5f, yC, 0},
56 {frameW * 0.5f, frameH * 0.5f, kBoardHalf});
57
58 frame_.upload(v, i);
59 }
60
61 // ---- File / rank labels engraved onto the frame top ----
62 // stb_easy_font emits a quad mesh in 2D font coords (origin top-left,
63 // +X right, +Y down). We map each label's quads onto the world-space
64 // frame surface so they rotate naturally with the camera.
65 {
66 struct EFV { float x, y, z; unsigned char c[4]; };
67 std::vector<Vertex> v;
68 std::vector<std::uint32_t> i;
69
70 const float scale = 0.045f; // ~0.45 unit tall glyph
71 const float labelY = 0.001f; // sits flush on frame top
72 const float edge = kBoardHalf + 0.30f; // mid of 0.6-wide frame strip
73
74 auto emit = [&](const char* text,
75 const glm::vec3& origin,
76 const glm::vec3& axisX,
77 const glm::vec3& axisZ) {
78 EFV buf[512];
79 int quads = stb_easy_font_print(0.0f, 0.0f,
80 const_cast<char*>(text),
81 nullptr, buf, sizeof(buf));
82 if (quads <= 0) return;
83 // Center on bounding box so the origin is the glyph center.
84 float minX = buf[0].x, maxX = buf[0].x;
85 float minY = buf[0].y, maxY = buf[0].y;
86 for (int q = 0; q < quads * 4; ++q) {
87 minX = std::min(minX, buf[q].x);
88 maxX = std::max(maxX, buf[q].x);
89 minY = std::min(minY, buf[q].y);
90 maxY = std::max(maxY, buf[q].y);
91 }
92 const float cx = (minX + maxX) * 0.5f;
93 const float cy = (minY + maxY) * 0.5f;
94
95 for (int q = 0; q < quads; ++q) {
96 std::uint32_t base = static_cast<std::uint32_t>(v.size());
97 for (int k = 0; k < 4; ++k) {
98 const EFV& fv = buf[q * 4 + k];
99 glm::vec3 p = origin
100 + axisX * ((fv.x - cx) * scale)
101 + axisZ * ((fv.y - cy) * scale);
102 v.push_back({ p, glm::vec3(0, 1, 0), glm::vec2(0) });
103 }
104 i.insert(i.end(), { base, base + 1, base + 2,
105 base, base + 2, base + 3 });
106 }
107 };
108
109 char t[2] = { 0, 0 };
110 // Files a..h on both Z edges, oriented to read from each side.
111 for (int f = 0; f < 8; ++f) {
112 const float x = (static_cast<float>(f) - 3.5f) * kSquareSize;
113 t[0] = static_cast<char>('a' + f);
114 emit(t, { x, labelY, edge }, { 1, 0, 0}, {0, 0, 1});
115 emit(t, { x, labelY, -edge }, {-1, 0, 0}, {0, 0, -1});
116 }
117 // Ranks 1..8 on both X edges.
118 for (int r = 0; r < 8; ++r) {
119 const float z = (3.5f - static_cast<float>(r)) * kSquareSize;
120 t[0] = static_cast<char>('1' + r);
121 emit(t, { edge, labelY, z }, {0, 0, -1}, { 1, 0, 0});
122 emit(t, { -edge, labelY, z }, {0, 0, 1}, {-1, 0, 0});
123 }
124
125 if (!v.empty()) labels_.upload(v, i);
126 }
127
128 clearHighlights();
129}

139void ChessBoard::highlightSquare(Square s, float strength) {
140 if (!s.valid()) return;
141 highlights_[index(s.file, s.rank)] = std::max(highlights_[index(s.file, s.rank)], strength);

144void ChessBoard::update(float dt) {
145 // Gentle exponential decay so highlights fade out softly.
146 const float decay = std::exp(-dt * 0.8f);
147 for (auto& h : highlights_) {
148 h *= decay;
149 if (h < 0.01f) h = 0.0f;
150 }
151}
