15// Procedurally generated 8x8 chess board. Each square is its own
16// little Mesh so that the renderer can pick which squares are
17// highlighted on a per-square basis.
18class ChessBoard {
19public:
20 ChessBoard();
21 void build();
22
23 // Mesh access:
24 const Mesh& squareMesh(int file, int rank) const;
25 const Mesh& frameMesh() const { return frame_; }
26 const Mesh& labelsMesh() const { return labels_; }
27
28 // Highlights - the renderer reads these every frame.
29 void clearHighlights();
30 void highlightSquare(Square s, float strength = 1.0f);
31
32 float highlight(int file, int rank) const {
33 return highlights_[index(file, rank)];
34 }
35
36 // Decay highlights with time so they pulse softly.
37 void update(float dt);
38
39private:
40 static int index(int f, int r) { return r * 8 + f; }
41
42 std::array<std::unique_ptr<Mesh>, 64> squares_;
43 std::array<float, 64> highlights_{};
44 Mesh frame_;
45 Mesh labels_;
46};
