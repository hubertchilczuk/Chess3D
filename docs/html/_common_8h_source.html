64constexpr float kSquareSize = 1.0f; // world units per square
65constexpr float kBoardHalf = 4.0f * kSquareSize;

67inline glm::vec3 squareToWorld(const Square& s) {
68 // Center each square; file -> X, rank -> Z (mirrored so White is at -Z)
69 float x = (static_cast<float>(s.file) - 3.5f) * kSquareSize;
70 float z = (3.5f - static_cast<float>(s.rank)) * kSquareSize;
71 return glm::vec3(x, 0.0f, z);
72}
