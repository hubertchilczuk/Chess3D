23bool Renderer::initialize() {
24 // ---- Shaders ----
25 const std::string shaderDir = CHESS3D_SHADERS_DIR;
26 if (!pieceShader_.loadFromFiles(shaderDir + "/phong.vert",
27 shaderDir + "/phong.frag")) {
28 return false;
29 }
30 if (!boardShader_.loadFromFiles(shaderDir + "/phong.vert",
31 shaderDir + "/phong.frag")) {
32 return false;
33 }
34
35 // ---- Textures ----
36 const std::string texDir = std::string(CHESS3D_RESOURCES_DIR) + "/textures";
37
38 auto tryLoadOrProc = [&](Texture& dst, const std::string& filename,
39 bool light, bool wood) {
40 std::string path = texDir + "/" + filename;
41 if (fs::exists(path) && dst.loadFromFile(path)) {
42 std::cout << "[Renderer] loaded texture: " << path << '\n';
43 return;
44 }
45 if (wood) dst.createProceduralWood (512, light);
46 else dst.createProceduralMarble(512, light);
47 };
48 tryLoadOrProc(lightWood_, "wood_light.png", true, true);
49 tryLoadOrProc(darkWood_, "wood_dark.png", false, true);
50 tryLoadOrProc(whiteMarble_, "marble_white.png", true, false);
51 tryLoadOrProc(blackMarble_, "marble_black.png", false, false);
52
53 // ---- Piece models ----
54 // Expects a single combined file in resources/ or resources/models/
55 // (no named groups) split into connected components by height.
56 const std::string resDir = CHESS3D_RESOURCES_DIR;
57 const std::string modelDir = resDir + "/models";
58
59 static constexpr const char* kCombinedExts[] = {
60 ".obj", ".fbx", ".gltf", ".glb", ".dae", ".blend"
61 };
62 std::string combinedPath;
63 for (const std::string& dir : { resDir, modelDir }) {
64 if (!fs::exists(dir) || !fs::is_directory(dir)) continue;
65 for (const auto& e : fs::directory_iterator(dir)) {
66 if (!e.is_regular_file()) continue;
67 std::string ext = e.path().extension().string();
68 for (char& c : ext) c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
69 bool isModelExt = false;
70 for (const char* ce : kCombinedExts) if (ext == ce) { isModelExt = true; break; }
71 if (!isModelExt) continue;
72
73 std::string stem = e.path().stem().string();
74 std::string lower = stem;
75 for (char& c : lower) c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
76 if (lower == "pawn" || lower == "knight" || lower == "bishop" ||
77 lower == "rook" || lower == "queen" || lower == "king") continue;
78
79 combinedPath = e.path().string();
80 break;
81 }
82 if (!combinedPath.empty()) break;
83 }
84
85 if (!combinedPath.empty()) {
86 // Heights sorted ascending: pawn < rook < knight < bishop < queen < king
87 static constexpr PieceType heightOrder[6] = {
88 PieceType::Pawn, PieceType::Rook, PieceType::Knight,
89 PieceType::Bishop, PieceType::Queen, PieceType::King
90 };
91 auto comps = Model::loadAndSplitByComponents(combinedPath);
92 if (comps.size() >= 6) {
93 std::cout << "[Renderer] split " << comps.size()
94 << " connected components from "
95 << fs::path(combinedPath).filename().string() << '\n';
96 for (int i = 0; i < 6; ++i)
97 pieceModels_[static_cast<int>(heightOrder[i])] = std::move(comps[i]);
98 } else {
99 std::cerr << "[Renderer] combined file produced only "
100 << comps.size() << " components (need 6) - "
101 "some piece types will be missing.\n";
102 }
103 }
104
105 // ---- Board geometry ----
106 board_.build();
107
108 // ---- Global GL state ----
109 glEnable(GL_DEPTH_TEST);
110 glDepthFunc(GL_LESS);
111 glEnable(GL_CULL_FACE);
112 glCullFace(GL_BACK);
113 glFrontFace(GL_CCW);
114 glEnable(GL_FRAMEBUFFER_SRGB);
115 glClearColor(0.06f, 0.07f, 0.09f, 1.0f);
116 return true;
117}

125void Renderer::uploadCommonUniforms(Shader& sh, const Camera& camera, float aspect) {
126 const glm::mat4 v = camera.view();
127 const glm::mat4 p = camera.projection(aspect);
128
129 sh.bind();
130 sh.setMat4("v", v);
131 sh.setMat4("p", p);
132 sh.setVec3("uViewPos", camera.position());
133 uploadLights(sh, camera);
134}

151void Renderer::drawBoard(const Camera& camera, const ChessGame& game, float aspect) {
152 Shader& sh = boardShader_;
153 uploadCommonUniforms(sh, camera, aspect);
154 sh.setInt ("uDiffuseTex", 0);
155 sh.setBool("uHasTexture", true);
156 sh.setVec3("uMaterialSpecular", glm::vec3(0.20f));
157 sh.setFloat("uMaterialShininess", 32.0f);
158
159 // For "last move" highlight - light up source & destination of the latest move.
160 Square hiFrom{-1,-1}, hiTo{-1,-1};
161 if (game.hasLastMove()) {
162 const Move& lm = game.lastMove();
163 hiFrom = lm.from;
164 hiTo = lm.to;
165 }
166
167 for (int r = 0; r < 8; ++r) {
168 for (int f = 0; f < 8; ++f) {
169 bool light = ((f + r) & 1) != 0;
170 if (light) lightWood_.bind(0);
171 else darkWood_ .bind(0);
172
173 glm::mat4 m(1.0f); // squares already in world space
174 glm::mat3 nm = glm::mat3(glm::transpose(glm::inverse(m)));
175
176 sh.setMat4("m", m);
177 sh.setMat3("uNormalMatrix", nm);
178 sh.setVec3("uMaterialDiffuse", glm::vec3(1.0f));
179 sh.setVec3("uTintColor", glm::vec3(1.0f));
180
181 float hl = board_.highlight(f, r);
182 // boost highlight for last-move squares
183 if ((Square{f, r} == hiFrom) || (Square{f, r} == hiTo)) {
184 hl = std::max(hl, 0.65f);
185 }
186 sh.setFloat("uHighlight", hl);
187
188 board_.squareMesh(f, r).draw();
189 }
190 }
191
192 // Frame (dark wood, no highlight)
193 darkWood_.bind(0);
194 glm::mat4 m(1.0f);
195 sh.setMat4("m", m);
196 sh.setMat3("uNormalMatrix", glm::mat3(1.0f));
197 sh.setVec3("uMaterialDiffuse", glm::vec3(0.55f));
198 sh.setVec3("uTintColor", glm::vec3(1.0f));
199 sh.setFloat("uHighlight", 0.0f);
200 board_.frameMesh().draw();
201
202 // Labels engraved on the frame top. Flat quads with normal +Y so they're
203 // double-sided-ish under typical lighting — disable culling just in case
204 // the per-edge basis flips winding.
205 sh.setBool("uHasTexture", false);
206 sh.setVec3("uMaterialDiffuse", glm::vec3(0.96f, 0.92f, 0.78f));
207 sh.setVec3("uMaterialSpecular", glm::vec3(0.05f));
208 sh.setFloat("uMaterialShininess", 8.0f);
209 sh.setMat4("m", m);
210 sh.setMat3("uNormalMatrix", glm::mat3(1.0f));
211 sh.setVec3("uTintColor", glm::vec3(1.0f));
212 sh.setFloat("uHighlight", 0.0f);
213 glDisable(GL_CULL_FACE);
214 board_.labelsMesh().draw();
215 glEnable(GL_CULL_FACE);
216}

218void Renderer::drawPieces(const Camera& camera, const ChessGame& game, float aspect) {
219 Shader& sh = pieceShader_;
220 uploadCommonUniforms(sh, camera, aspect);
221 sh.setInt ("uDiffuseTex", 0);
222 sh.setBool("uHasTexture", true);
223 sh.setVec3("uMaterialSpecular", glm::vec3(0.55f));
224 sh.setFloat("uMaterialShininess", 64.0f);
225
226 // Highlight the piece currently animating (the most recent mover).
227 const ChessPiece* recentMover = nullptr;
228 for (const auto& up : game.pieces()) {
229 if (up->animator().isPlaying()) {
230 recentMover = up.get();
231 break;
232 }
233 }
234
235 for (const auto& up : game.pieces()) {
236 const ChessPiece& p = *up;
237 if (p.captured()) continue;
238 Model* model = pieceModels_[static_cast<int>(p.type())].get();
239 if (!model) continue;
240
241 if (p.color() == Color::White) whiteMarble_.bind(0);
242 else blackMarble_.bind(0);
243
244 glm::vec3 pos = p.animator().currentPosition();
245 glm::mat4 m(1.0f);
246 m = glm::translate(m, pos);
247 if (p.type() == PieceType::Knight) {
248 // Orient knights so they face the opposing side.
249 float rot = (p.color() == Color::White) ? 0.0f : glm::pi<float>();
250 m = glm::rotate(m, rot, glm::vec3(0, 1, 0));
251 }
252 // Slight uniform scale so pieces fit comfortably inside a square.
253 m = glm::scale(m, glm::vec3(0.95f));
254 // Apply per-model normalisation last (identity for procedural pieces,
255 // axis-fix + recenter + height-fit for imported Blender/FBX meshes).
256 m = m * model->normalizationTransform();
257
258 glm::mat3 nm = glm::mat3(glm::transpose(glm::inverse(m)));
259 sh.setMat4("m", m);
260 sh.setMat3("uNormalMatrix", nm);
261 sh.setVec3("uMaterialDiffuse", glm::vec3(1.0f));
262
263 // Color tint
264 glm::vec3 tint = (p.color() == Color::White)
265 ? glm::vec3(1.00f, 0.98f, 0.92f)
266 : glm::vec3(0.45f, 0.40f, 0.38f);
267 sh.setVec3("uTintColor", tint);
268
269 float hl = (&p == recentMover) ? 0.45f : 0.0f;
270 sh.setFloat("uHighlight", hl);
271
272 model->draw();
273 }
274}

276void Renderer::render(const Camera& camera, const ChessGame& game, float /*time*/) {
277 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
278 if (height_ <= 0) return;
279 float aspect = static_cast<float>(width_) / static_cast<float>(height_);
280
281 drawBoard (camera, game, aspect);
282 drawPieces(camera, game, aspect);
283}
