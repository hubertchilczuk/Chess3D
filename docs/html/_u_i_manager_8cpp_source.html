23bool UIManager::initialize(GLFWwindow* window) {
24 IMGUI_CHECKVERSION();
25 ImGui::CreateContext();
26 ImGui::StyleColorsDark();
27
28 ImGuiIO& io = ImGui::GetIO();
29 io.IniFilename = nullptr; // no imgui.ini side files
30 io.ConfigFlags |= ImGuiConfigFlags_NavEnableKeyboard;
31
32 if (!ImGui_ImplGlfw_InitForOpenGL(window, true)) return false;
33 if (!ImGui_ImplOpenGL3_Init("#version 330 core")) return false;
34
35 std::snprintf(pgnPathBuffer_, sizeof(pgnPathBuffer_),
36 "%s/pgn/sample_game.pgn", CHESS3D_RESOURCES_DIR);
37 initialized_ = true;
38 return true;
39}

66void UIManager::drawControlPanel(ChessGame& game, Renderer& renderer, Camera& camera, float fps) {
67 ImGui::SetNextWindowPos (ImVec2(10, 10), ImGuiCond_FirstUseEver);
68 ImGui::SetNextWindowSize(ImVec2(320, 420), ImGuiCond_FirstUseEver);
69 ImGui::Begin("Chess3D - Control");
70
71 ImGui::Text("FPS: %.1f", static_cast<double>(fps));
72
73 ImGui::Separator();
74 ImGui::TextColored(ImVec4(0.9f, 0.85f, 0.4f, 1.0f), "Game");
75 if (!game.pgnHeader("White").empty()) {
76 ImGui::Text("White : %s", game.pgnHeader("White").c_str());
77 }
78 if (!game.pgnHeader("Black").empty()) {
79 ImGui::Text("Black : %s", game.pgnHeader("Black").c_str());
80 }
81 if (!game.pgnHeader("Event").empty()) {
82 ImGui::Text("Event : %s", game.pgnHeader("Event").c_str());
83 }
84
85 ImGui::Separator();
86 ImGui::TextColored(ImVec4(0.9f, 0.85f, 0.4f, 1.0f), "Playback");
87 ImGui::Text("Move %d / %d", game.currentMoveIndex(), game.totalMoves());
88
89 if (game.isPlaying()) {
90 if (ImGui::Button("Pause")) game.pause();
91 } else {
92 if (ImGui::Button("Play")) game.play();
93 }
94 ImGui::SameLine();
95 if (ImGui::Button("<< Prev")) game.stepBackward();
96 ImGui::SameLine();
97 if (ImGui::Button("Next >>")) game.stepForward();
98 ImGui::SameLine();
99 if (ImGui::Button("Reset")) { game.resetToStartingPosition(); }
100
101 float spd = game.speed();
102 if (ImGui::SliderFloat("Speed (mv/s)", &spd, 0.1f, 4.0f, "%.2f")) {
103 game.setSpeed(spd);
104 }
105
106 ImGui::Separator();
107 ImGui::TextColored(ImVec4(0.9f, 0.85f, 0.4f, 1.0f), "PGN");
108 ImGui::InputText("Path", pgnPathBuffer_, sizeof(pgnPathBuffer_));
109 if (ImGui::Button("Load PGN")) {
110 if (game.loadPGNFile(pgnPathBuffer_)) {
111 statusMessage_ = "Loaded " +
112 std::filesystem::path(pgnPathBuffer_).filename().string();
113 } else {
114 statusMessage_ = "Failed to load PGN";
115 }
116 }
117 ImGui::SameLine();
118 if (ImGui::Button("Reload sample")) {
119 std::string p = std::string(CHESS3D_RESOURCES_DIR) + "/pgn/sample_game.pgn";
120 std::strncpy(pgnPathBuffer_, p.c_str(), sizeof(pgnPathBuffer_) - 1);
121 if (game.loadPGNFile(p)) statusMessage_ = "Loaded sample_game.pgn";
122 }
123 if (!statusMessage_.empty()) {
124 ImGui::TextWrapped("%s", statusMessage_.c_str());
125 }
126
127 ImGui::Separator();
128 ImGui::TextColored(ImVec4(0.9f, 0.85f, 0.4f, 1.0f), "Lighting");
129 ImGui::DragFloat3("Dir light direction", &renderer.dirLight().direction.x, 0.01f, -1.0f, 1.0f);
130 ImGui::ColorEdit3("Dir diffuse", &renderer.dirLight().diffuse.x);
131 ImGui::DragFloat3("Point light pos", &renderer.pointLight().position.x, 0.05f);
132 ImGui::ColorEdit3("Point diffuse", &renderer.pointLight().diffuse.x);
133
134 ImGui::Separator();
135 ImGui::TextColored(ImVec4(0.9f, 0.85f, 0.4f, 1.0f), "Camera");
136 ImGui::Text("Pos: %.1f, %.1f, %.1f",
137 static_cast<double>(camera.position().x),
138 static_cast<double>(camera.position().y),
139 static_cast<double>(camera.position().z));
140 ImGui::DragFloat("Move speed", &camera.moveSpeed, 0.1f, 0.5f, 30.0f);
141 ImGui::DragFloat("Mouse sens", &camera.mouseSensitivity, 0.01f, 0.01f, 1.0f);
142
143 ImGui::End();
144}

146void UIManager::drawMoveList(ChessGame& game) {
147 ImGui::SetNextWindowPos (ImVec2(10, 440), ImGuiCond_FirstUseEver);
148 ImGui::SetNextWindowSize(ImVec2(320, 250), ImGuiCond_FirstUseEver);
149 ImGui::Begin("Moves");
150
151 const auto& moves = game.moves();
152 if (ImGui::BeginTable("moves_table", 3,
153 ImGuiTableFlags_RowBg | ImGuiTableFlags_Borders |
154 ImGuiTableFlags_ScrollY)) {
155 ImGui::TableSetupColumn("#", ImGuiTableColumnFlags_WidthFixed, 40.0f);
156 ImGui::TableSetupColumn("White");
157 ImGui::TableSetupColumn("Black");
158 ImGui::TableHeadersRow();
159
160 int rows = (static_cast<int>(moves.size()) + 1) / 2;
161 for (int row = 0; row < rows; ++row) {
162 ImGui::TableNextRow();
163 ImGui::TableSetColumnIndex(0);
164 ImGui::Text("%d.", row + 1);
165
166 int wIdx = row * 2;
167 int bIdx = row * 2 + 1;
168 ImGui::TableSetColumnIndex(1);
169 if (wIdx < static_cast<int>(moves.size())) {
170 bool current = (game.currentMoveIndex() == wIdx + 1);
171 if (current) ImGui::TextColored(ImVec4(1, 0.85f, 0.3f, 1), "%s", moves[wIdx].san.c_str());
172 else ImGui::TextUnformatted(moves[wIdx].san.c_str());
173 }
174 ImGui::TableSetColumnIndex(2);
175 if (bIdx < static_cast<int>(moves.size())) {
176 bool current = (game.currentMoveIndex() == bIdx + 1);
177 if (current) ImGui::TextColored(ImVec4(1, 0.85f, 0.3f, 1), "%s", moves[bIdx].san.c_str());
178 else ImGui::TextUnformatted(moves[bIdx].san.c_str());
179 }
180 }
181 ImGui::EndTable();
182 }
183 ImGui::End();
184}
